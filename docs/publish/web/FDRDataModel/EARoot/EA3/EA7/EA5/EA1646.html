viewdef SELECT rec.fund_id,
    rec.fund_code,
    rec.fund_description,
    rec.fund_is_federal,
    rec.fund_fiscal_year,
    rec.fund_is_valid_for_override,
    rec.created_at,
    rec.created_by_user_id,
    rec.created_by_user_sub,
    rec.created_by_user_first_name,
    rec.created_by_user_last_name,
    rec.created_by_user_email,
    rec.updated_at,
    rec.updated_by_user_id,
    rec.updated_by_user_sub,
    rec.updated_by_user_first_name,
    rec.updated_by_user_last_name,
    rec.updated_by_user_email,
    rec.start_date,
    rec.end_date,
    rec.ods_id,
    rec.ods_meta,
    rec.ods_created_by,
    rec.ods_created_at
   FROM (funds rec
     JOIN ( SELECT i.fund_id,
            max(i.ods_created_at) AS max_version
           FROM funds i
          GROUP BY i.fund_id) m ON (((rec.fund_id = m.fund_id) AND (rec.ods_created_at = m.max_version))))
  ORDER BY rec.fund_id;
