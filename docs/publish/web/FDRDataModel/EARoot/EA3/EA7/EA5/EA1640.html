viewdef SELECT rec.distribution_line_index_id,
    rec.distribution_line_index_code,
    rec.created_at,
    rec.created_by_user_id,
    rec.created_by_user_sub,
    rec.created_by_user_first_name,
    rec.created_by_user_last_name,
    rec.created_by_user_email,
    rec.updated_at,
    rec.updated_by_user_id,
    rec.updated_by_user_sub,
    rec.updated_by_user_first_name,
    rec.updated_by_user_last_name,
    rec.updated_by_user_email,
    rec.start_date,
    rec.end_date,
    rec.ods_id,
    rec.ods_meta,
    rec.ods_created_by,
    rec.ods_created_at
   FROM (distribution_line_indexes rec
     JOIN ( SELECT i.distribution_line_index_id,
            max(i.ods_created_at) AS max_version
           FROM distribution_line_indexes i
          GROUP BY i.distribution_line_index_id) m ON (((rec.distribution_line_index_id = m.distribution_line_index_id) AND (rec.ods_created_at = m.max_version))))
  ORDER BY rec.distribution_line_index_id;
