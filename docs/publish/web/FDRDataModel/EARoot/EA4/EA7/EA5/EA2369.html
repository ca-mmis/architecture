viewdef SELECT b.transaction_request_form_id,
    b.transaction_request_form_agency,
    b.transaction_request_form_type,
    b.transaction_request_form_description,
    b.transaction_request_form_division,
    b.transaction_request_form_non_program_type,
    b.transaction_request_form_title,
    b.transaction_request_form_primary_bucket_ids,
    b.transaction_request_form_secondary_bucket_ids,
    b.transaction_request_form_fdr_appropriation_reference_id,
    b.transaction_request_form_fdr_appropriation_reference_code,
    b.created_at,
    b.created_by_user_id,
    b.created_by_user_sub,
    b.created_by_user_first_name,
    b.created_by_user_last_name,
    b.created_by_user_email,
    b.updated_at,
    b.updated_by_user_id,
    b.updated_by_user_sub,
    b.updated_by_user_first_name,
    b.updated_by_user_last_name,
    b.updated_by_user_email,
    b.start_date,
    b.end_date,
    b.ods_id,
    b.ods_meta,
    b.ods_created_by,
    b.ods_created_at
   FROM (transaction_request_forms b
     JOIN ( SELECT i.transaction_request_form_id,
            max(i.ods_created_at) AS max_version
           FROM transaction_request_forms i
          GROUP BY i.transaction_request_form_id) m ON (((b.transaction_request_form_id = m.transaction_request_form_id) AND (b.ods_created_at = m.max_version))))
  ORDER BY b.transaction_request_form_id;
