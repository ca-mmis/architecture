viewdef SELECT b.bank_account_id,
    b.bank_account_bank_account,
    b.created_by,
    b.updated_by,
    b.created_at,
    b.updated_at,
    b.start_date,
    b.end_date,
    b.id,
    b.ods_meta,
    b.ods_version,
    b.ods_created_by,
    b.ods_created_at
   FROM (bank_accounts b
     JOIN ( SELECT i.bank_account_id,
            max(i.ods_created_at) AS max_version
           FROM bank_accounts i
          GROUP BY i.bank_account_id) m ON (((b.bank_account_id = m.bank_account_id) AND (b.ods_created_at = m.max_version))))
  ORDER BY b.bank_account_id;
