viewdef SELECT rec.fund_code_id,
    rec.fund_code_code,
    rec.fund_code_description,
    rec.fund_code_reporting_structure,
    rec.fund_code_account_id,
    rec.fund_code_chapter_advances,
    rec.fund_code_enactment_year,
    rec.created_by,
    rec.updated_by,
    rec.created_at,
    rec.updated_at,
    rec.start_date,
    rec.end_date,
    rec.id,
    rec.ods_meta,
    rec.ods_version,
    rec.ods_created_by,
    rec.ods_created_at
   FROM (fund_codes rec
     JOIN ( SELECT i.fund_code_id,
            max(i.ods_created_at) AS max_version
           FROM fund_codes i
          GROUP BY i.fund_code_id) m ON (((rec.fund_code_id = m.fund_code_id) AND (rec.ods_created_at = m.max_version))))
  ORDER BY rec.fund_code_id;
