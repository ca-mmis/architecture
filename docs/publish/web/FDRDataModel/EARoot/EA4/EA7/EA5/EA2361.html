viewdef SELECT rec.program_type_id,
    rec.program_type_code,
    rec.program_type_description,
    rec.program_type_summary_program_type_id,
    rec.program_type_fiscal_appropriation_ref,
    rec.program_type_program,
    rec.program_type_element,
    rec.program_type_component,
    rec.created_at,
    rec.created_by_user_id,
    rec.created_by_user_sub,
    rec.created_by_user_first_name,
    rec.created_by_user_last_name,
    rec.created_by_user_email,
    rec.updated_at,
    rec.updated_by_user_id,
    rec.updated_by_user_sub,
    rec.updated_by_user_first_name,
    rec.updated_by_user_last_name,
    rec.updated_by_user_email,
    rec.start_date,
    rec.end_date,
    rec.ods_id,
    rec.ods_meta,
    rec.ods_created_by,
    rec.ods_created_at
   FROM (program_types rec
     JOIN ( SELECT i.program_type_id,
            max(i.ods_created_at) AS max_version
           FROM program_types i
          GROUP BY i.program_type_id) m ON (((rec.program_type_id = m.program_type_id) AND (rec.ods_created_at = m.max_version))))
  ORDER BY rec.program_type_id;
