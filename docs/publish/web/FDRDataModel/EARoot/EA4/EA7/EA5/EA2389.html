viewdef SELECT rec.service_category_id,
    rec.service_category_description,
    rec.service_category_line_number,
    rec.service_category_requires_feeder_form,
    rec.created_by,
    rec.updated_by,
    rec.created_at,
    rec.updated_at,
    rec.start_date,
    rec.end_date,
    rec.id,
    rec.ods_meta,
    rec.ods_version,
    rec.ods_created_by,
    rec.ods_created_at
   FROM (service_categories rec
     JOIN ( SELECT i.service_category_id,
            max(i.ods_created_at) AS max_version
           FROM service_categories i
          GROUP BY i.service_category_id) m ON (((rec.service_category_id = m.service_category_id) AND (rec.ods_created_at = m.max_version))))
  ORDER BY rec.service_category_id;
