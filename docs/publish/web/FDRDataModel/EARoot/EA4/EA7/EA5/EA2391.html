viewdef SELECT rec.summary_program_type_id,
    rec.summary_program_type_description,
    rec.created_by,
    rec.created_at,
    rec.updated_at,
    rec.updated_by,
    rec.start_date,
    rec.end_date,
    rec.id,
    rec.ods_meta,
    rec.ods_version,
    rec.ods_created_by,
    rec.ods_created_at
   FROM (summary_program_types rec
     JOIN ( SELECT i.summary_program_type_id,
            max(i.ods_created_at) AS max_version
           FROM summary_program_types i
          GROUP BY i.summary_program_type_id) m ON (((rec.summary_program_type_id = m.summary_program_type_id) AND (rec.ods_created_at = m.max_version))))
  ORDER BY rec.summary_program_type_id;
