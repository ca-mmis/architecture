viewdef SELECT rec.supplier_id,
    rec.supplier_code,
    rec.supplier_name,
    rec.supplier_fein,
    rec.created_at,
    rec.created_by_user_id,
    rec.created_by_user_sub,
    rec.created_by_user_first_name,
    rec.created_by_user_last_name,
    rec.created_by_user_email,
    rec.updated_at,
    rec.updated_by_user_id,
    rec.updated_by_user_sub,
    rec.updated_by_user_first_name,
    rec.updated_by_user_last_name,
    rec.updated_by_user_email,
    rec.start_date,
    rec.end_date,
    rec.ods_id,
    rec.ods_meta,
    rec.ods_created_by,
    rec.ods_created_at
   FROM (suppliers rec
     JOIN ( SELECT i.supplier_id,
            max(i.ods_created_at) AS max_version
           FROM suppliers i
          GROUP BY i.supplier_id) m ON (((rec.supplier_id = m.supplier_id) AND (rec.ods_created_at = m.max_version))))
  ORDER BY rec.supplier_id;
