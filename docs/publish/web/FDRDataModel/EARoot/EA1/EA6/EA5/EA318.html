viewdef SELECT rec.fund_ratio_id,
    rec.fund_ratio_federal_percentage,
    rec.fund_ratio_non_federal_percentage,
    rec.fund_ratio_is_service_category_required,
    rec.start_date,
    rec.end_date,
    rec.created_by,
    rec.created_at,
    rec.updated_at,
    rec.updated_by,
    rec.id,
    rec.ods_meta,
    rec.ods_version,
    rec.ods_created_by,
    rec.ods_created_at
   FROM (fund_ratios rec
     JOIN ( SELECT i.fund_ratio_id,
            max(i.ods_created_at) AS max_version
           FROM fund_ratios i
          GROUP BY i.fund_ratio_id) m ON (((rec.fund_ratio_id = m.fund_ratio_id) AND (rec.ods_created_at = m.max_version))))
  ORDER BY rec.fund_ratio_id;
